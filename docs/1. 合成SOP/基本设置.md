## 如何插入一段代码

以下是将it曲线转化为ic数据的matlab代码

```matlab
% ittoic函数版本升级，2025年3月8日
function current= ittoic(itdata,concentration,stoptime,win)
% itdata 就是导入的第一列是时间，剩下的列是电流
% concentration 是每次实际测试的浓度
% stoptime 是每次暂停的时间
% win 是从暂停时间往前数几个数据
ts = itdata(:,1);
it = itdata(:,2:end);
m = width(it); % it曲线的条数m
n = height(stoptime); % 需要计算的电流点数
C = mycolor(1,m); % 有曲线条数的颜色
%% 寻找加液点在ts上的位置矩阵
position = zeros(n,1);
for ii = 1:n
    position(ii) = find(ts == stoptime(ii));
end

%% 先把所有样品的i-t曲线做出来看一看：
figure
subplot 121
for ii=1:m
    plot(ts,it(:,ii),'LineWidth',1,'Color',C(ii,:));
    hold on
end
% 把测试点标在第一条i-t曲线上：
scatter(stoptime, it(position),'r','linewidth',2) 
grid on
title('raw data')
xlabel('time (s)')
ylabel('current (nA)')
set(gca,'FontSize',12)

%% 直接计算不同i-t曲线提取值的均值
current = zeros(n,m);
for jj = 1:m
    trailtemp = zeros(win,n);
    currenttemp = it(:,jj);
    for ii = 1:n
        trailtemp(:,ii) = filloutliers(currenttemp(position(ii)-(win-1):position(ii)), 'linear');
    end
    currenttemp = zeros(n,1);
    for ii = 1:n
        currenttemp(ii) = mean(trailtemp(:,ii));
    end
    current(:,jj) = currenttemp;
end
%% 做出所有的IC图像
subplot 122
hold on
for i = 1:m
    plot(concentration,current(:,i),'-o','LineWidth',1.5,'Color',C(i,:));
    grid on
end
box on
title('i-c data plot')
xlabel('concentration (mM)')
ylabel('current (nA)')
set(gca,'FontSize',12)

%% 调整不同的ylimit
nn = height(current);
current_max = max(current(nn,:));
yl = ceil(current_max / 10) * 10;
subplot 121
ylim([0 yl]);
subplot 122
ylim([0 yl]);
end
```
